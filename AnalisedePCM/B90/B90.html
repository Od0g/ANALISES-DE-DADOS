import matplotlib.pyplot as plt
import seaborn as sns

# Colunas para datatime
df["Data Incio"] = pd.to_datetime(df["Data Incio"])
df["Data Final"] = pd.to_datetime(df["Data Final"])

# Colunas auxiliares
df["Ano"] = df["Data Incio"].dt.year
df["Ms"] = df["Data Incio"].dt.month

# Colunas com Periodos
dados_jul_dez = df[df["Ms"].between(7, 12)]
dados_jan_todos = df[df["Ms"] == 1]
dados_problemas_jan_jun = df[df["Ms"].between(1, 6)]

# Contagem de ocorrncias por mquina para determinar a mais e menos problemtica
maquina_mais_problemas = df["Mquina"].value_counts().idxmax()
maquina_menos_problemas = df["Mquina"].value_counts().idxmin()


# Grfico de pizza - Distribuio de problemas por mquina
plt.figure(figsize=(8, 8))
df["Mquina"].value_counts().plot.pie(autopct='%1.1f%%', cmap="Set3", startangle=90)
plt.title("Distribuio de Problemas por Mquina")
plt.ylabel("")
plt.show()

# Grfico de barras horizontais - Contagem de problemas por mquina
plt.figure(figsize=(10, 6))
df["Mquina"].value_counts().plot(kind="barh", color="skyblue", edgecolor="black")
plt.xlabel("Quantidade de Problemas")
plt.ylabel("Mquina")
plt.title("Quantidade de Problemas por Mquina")
plt.show()

# Grfico de barras verticais - Problemas ao longo dos meses
plt.figure(figsize=(12, 6))
sns.countplot(x=df["Ms"], palette="viridis", edgecolor="black")
plt.xlabel("Ms")
plt.ylabel("Quantidade de Problemas")
plt.title("Problemas Registrados ao Longo dos Meses")
plt.show()

# Grfico de linearidade - Evoluo dos problemas ao longo do tempo
problemas_por_mes = df.groupby(["Ano", "Ms"]).size().reset_index(name="Contagem")
plt.figure(figsize=(12, 6))
sns.lineplot(x=problemas_por_mes.index, y=problemas_por_mes["Contagem"], marker="o", color="red")
plt.xlabel("Tempo (Ano-Ms)")
plt.ylabel("Quantidade de Problemas")
plt.title("Evoluo dos Problemas ao Longo do Tempo")
plt.grid(True)
plt.show()
